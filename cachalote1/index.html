let whale = document.getElementById('whale'); let gravity = 0.8; // Gravedad que hace que el cachalote caiga let jumpStrength = -15; // Fuerza del salto let velocity = 0; // Velocidad inicial let isJumping = false; // Variable para saber si el cachalote está saltando let groundLevel = window.innerHeight - whale.offsetHeight; function gameLoop() { if (isJumping) { velocity += gravity; whale.style.bottom = (parseInt(whale.style.bottom) + velocity) + 'px'; // Si el cachalote toca el suelo, se detiene el salto if (parseInt(whale.style.bottom) >= groundLevel) { whale.style.bottom = groundLevel + 'px'; isJumping = false; velocity = 0; } } requestAnimationFrame(gameLoop); } document.addEventListener('keydown', function(event) { if (event.code === 'Space' && !isJumping) { isJumping = true; velocity = jumpStrength; } }); // Inicialización whale.style.bottom = groundLevel + 'px'; gameLoop();
